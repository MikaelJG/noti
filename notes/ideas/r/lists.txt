### A list in R can contain many different data types inside it. A list is a collection of data which is ordered and changeable
### To create a list, use the list() function

# List of strings
thislist <- list("apple", "banana", "cherry")

# Print the list
thislist

# List length to know how many items are in the list
thislist <- list("apple", "banana", "cherry")
length(thislist)

# Add items to list
thislist <- list("apple", "banana", "cherry")
append(thislist, "orange")

# Add item at a specified location
thislist <- list("apple", "banana", "cherry")
append(thislist, "orange", after = 2)		# To add an item to the right of a specified index, add "after=index number" in the append() function

# Combine lists
list1 <- list("a", "b", "c")
list2 <- list(1,2,3)
list3 <- c(list1,list2)
list3
