let n = 0

function numberFormat(n) {
    return n.toString().padStart(2, '0')
}

function render() {
    const title = <h1 className="title" id="title">
        Bonjour <span>{numberFormat(n)}</span>
        </h1>
    ) 

    // brancher au dom avec ReactDOM
    ReactDOM.render(title, document.querySelector('#app')
}
