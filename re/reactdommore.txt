let n = 0

function render() {
    const title = React.createElement('h1', {},
    'Bonjour',
    React.createElement('span', {}, n)
    ) 

    // brancher au dom avec ReactDOM
    ReactDOM.render(title, document.querySelector('#app')

}

render()

window.setInterval(() => {
    n++
    render()
}, 1000)

<body>
    <div id="app"></div>
</body>
