class ManualIncrementor extends React.Component {
    constructor (props) {
        super(props)
        this.state = {n: 0}
    }

    increment (e) {
        console.log(e)
        this.setState((state, props) => ({n: state.n + 1}))
    }

    render () {
        return <div>Valeur : {this.state.n} <button onClick=(this.increment.bind(this))>Incrementor</button></div>
    }
}
